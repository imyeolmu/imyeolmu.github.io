### RefType
- 참조형
  new 연산자로 객체를 생성해서 객체를 사용하는 타입이다
  

 ```java

package java_study;

	
public class JAVA0509_1{
    //기본 생성자는 컴파일러에서 생성해준다
	public JAVA0509_1() {
	}
    //참조형(RefType)
	public void refType() {
        //너무 자주 사용하기 때문에 기본자료형처럼 문자열 값을 직접할당하게 해준다
		String str1 = "Hello";
		System.out.println("str1:"+str1);
		
		String str2 = new String("Hello!");
		System.out.println("str2:"+str2);
		//문자열의 값을 더하기 연산자로 결합하면 결과는 문자열 이다
		String str3 = "java~";
		System.out.println(str1+ str3);
    }
	
	
	public static void main(String arg[]) {
		JAVA0509_1 dt =new JAVA0509_1();
		dt.refType();
		
		
	
		
	}
}	

```

```java

str1:Hello
str2:Hello!
Hellojava~

```
```java

package java_study;

	
public class JAVA0509_1{
	public void refType() {
	
		int m = 10;
		int n = 30;

		System.out.println(m+n+" <-합");
		System.out.println("합 ->"+m+n);
		//합 -> 40출력해보기
		System.out.println("합->"+(m+n));
	}
	
	
	public static void main(String arg[]) {
		JAVA0509_1 dt =new JAVA0509_1();
		dt.refType();
		
		
	
		
	}
}	


```
```java
40 <-합
합 ->1030
합->40
```
