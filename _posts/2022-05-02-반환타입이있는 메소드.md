### 반환 타입이 있는 메소드


```java
package day01;

public class Hello {
    public Hello() {
        system.out.println("생성자입니다");
    }


//클래스 메소드 반환타입이 int인 메소드
    public static int aa() {
        System.out.println("aa()메소드");

        return 1000;
        return "test";

    } 


    public static String bb(){
        system.out.println("bb()메소드");
        
        return"500";
    }


//non-static 메소드
    public void foo(){
        System.out.println("foo()메소드");
    }
    public int cc{
        System.out.println("cc()메소드");
        return "300"; //반환타입 이 void가 아닌경우 return타입이 필요하다

    }


    //리턴되는 값의 타입과 변수의 타입이 일치해야한다
    public static void main(String arg[]) {
        int money = Hello.aa(); //클래스 메소드를 호출-반환타입이 있을경우
        System.out.println("money:" + money);

        String money2 = Hello.bb();
        System.out.println("money2:" + money2);


        
        Hello hello = new Hello();
        hello.foo();
      //  반환타입이 있는 non-static 메소드 호출
        int money3 = hello.cc();
        System.out.println(money3 + "만원");

    }


}
```
